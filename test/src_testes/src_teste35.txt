.global _boot
.text

_boot:                    /* x0  = 0    0x000 */
    /* Test ADDI */
    addi x1 , x0,   1000  /* x1  = 1000 0x3E8 */
    addi x2 , x0,   2000  /* x2  = 3000 0xBB8 */
    addi x3,  x0,      0
    
    addi x1, x1, 100
    addi x3, x3,   1
    blt x1, x2, -8

    addi x4, x0, 1024
    addi x4, x4, 1024
    
    sw x1, 0(x4)
    sw x2, 4(x4)
    sw x3, 8(x4)
    
.data
variable:
	.word 0xdeadbeef
